/*
  Given a table and column on which a running total is to be calculated, 
  invoke fxGenerateRunningTotalColumn() to return a list of values for the new 
  running total column. Then invoke Table.FromColumns() using the list of lists
  generated by GetTableColumns() and the list of values returned by invoking 
  fxGenerateRunningTotalColumn(), and as the final argument, a list of the names 
  of the columns from the specified table combined with new running total column 
  named "Running Total".
*/

(source_table as table, data_column as list) =>

let
    RT_Column = fxGenerateRunningTotalColumn(data_column),
    Source_Table_Columns = fxGetTableColumns(source_table),
    Source_Table_Column_Names = Table.ColumnNames(source_table),  
    Output_Table = Table.FromColumns(Source_Table_Columns & {RT_Column}, 
                                     Source_Table_Column_Names & {"Running Total"})
in
    Output_Table
